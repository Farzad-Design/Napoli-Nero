'use client';

import { useRef } from 'react';
import Image from 'next/image';
import { motion } from 'framer-motion';
import { gsap } from 'gsap';
import { ScrollTrigger } from 'gsap/ScrollTrigger';
import { useGSAP } from '@gsap/react';

gsap.registerPlugin(ScrollTrigger);

export default function About() {
  const sectionRef  = useRef<HTMLDivElement>(null);
  const imageRef    = useRef<HTMLDivElement>(null);
  const contentRef  = useRef<HTMLDivElement>(null);

  useGSAP(() => {
    const section = sectionRef.current;
    if (!section) return;

    /* Parallax on the image */
    gsap.fromTo(
      imageRef.current,
      { y: 50 },
      {
        y: -50,
        ease: 'none',
        scrollTrigger: {
          trigger:  section,
          start:    'top bottom',
          end:      'bottom top',
          scrub:    1.5,
        },
      },
    );

    /* Content reveal */
    const items = contentRef.current?.querySelectorAll('.reveal-item');
    if (items) {
      gsap.fromTo(
        items,
        { y: 50, opacity: 0 },
        {
          y: 0, opacity: 1, duration: 0.8,
          stagger: 0.15,
          ease: 'power2.out',
          scrollTrigger: {
            trigger:  contentRef.current,
            start:    'top 75%',
            toggleActions: 'play none none reverse',
          },
        },
      );
    }

    /* Image reveal */
    gsap.fromTo(
      '.about-image-container',
      { scale: 0.92, opacity: 0, x: -40 },
      {
        scale: 1, opacity: 1, x: 0, duration: 1.0,
        ease: 'power2.out',
        scrollTrigger: {
          trigger:  section,
          start:    'top 70%',
          toggleActions: 'play none none reverse',
        },
      },
    );
  }, { scope: sectionRef });

  return (
    <section
      ref={sectionRef}
      id="about"
      className="relative py-28 bg-[var(--bg-primary)] overflow-hidden"
    >
      {/* Ambient glow */}
      <div className="absolute top-0 right-0 w-[500px] h-[500px] rounded-full bg-ember-500/[0.04] blur-[120px] pointer-events-none" />

      <div className="container-site">
        <div className="grid lg:grid-cols-2 gap-16 lg:gap-24 items-center">

          {/* ── Image side ─────────────────────────────── */}
          <div className="about-image-container relative" ref={imageRef}>
            {/* Outer decorative frame */}
            <div
              className="absolute -inset-4 rounded-3xl border border-ember-500/[0.12]"
              style={{ background: 'linear-gradient(135deg, rgba(255,107,0,0.04) 0%, transparent 60%)' }}
            />

            {/* Image wrapper */}
            <div className="relative w-full aspect-[4/5] max-w-[480px] rounded-2xl overflow-hidden"
              style={{ boxShadow: '0 20px 80px rgba(0,0,0,0.6), 0 0 0 1px var(--border)' }}
            >
              <Image
                src="https://images.unsplash.com/photo-1555396273-367ea4eb4db5?w=800&q=80"
                alt="Welcome to Napoli Nero restaurant interior"
                fill
                sizes="(max-width: 768px) 100vw, 50vw"
                className="object-cover"
              />
              {/* Inner overlay */}
              <div className="absolute inset-0 bg-gradient-to-t from-noir-950/50 via-transparent to-transparent" />

              {/* Floating counter badge */}
              <motion.div
                initial={{ opacity: 0, scale: 0.8 }}
                whileInView={{ opacity: 1, scale: 1 }}
                viewport={{ once: true }}
                transition={{ delay: 0.6, duration: 0.5 }}
                className="absolute bottom-6 right-6 glass rounded-xl p-3.5 flex items-center gap-3"
              >
                <div className="w-9 h-9 rounded-lg bg-ember-500/25 flex items-center justify-center">
                  <span className="text-lg">🔥</span>
                </div>
                <div>
                  <div className="text-white font-display font-bold text-xl leading-none">15+</div>
                  <div className="text-[var(--text-muted)] text-[0.6rem] tracking-wider mt-0.5">Years of Excellence</div>
                </div>
              </motion.div>
            </div>

            {/* Decorative corner */}
            <div className="absolute -bottom-6 -left-6 w-32 h-32 border-b-2 border-l-2 border-ember-500/20 rounded-bl-3xl" />
          </div>

          {/* ── Content side ───────────────────────────── */}
          <div ref={contentRef} className="flex flex-col gap-6">

            <div className="reveal-item flex items-center gap-3">
              <div className="w-8 h-[1.5px] bg-ember-500" />
              <span className="text-ember-400 text-[0.7rem] font-medium tracking-[0.25em] uppercase font-body">
                Our Story
              </span>
            </div>

            <h2 className="reveal-item heading-section text-white">
              Welcome to Our<br />
              <span className="text-ember-500 italic">Restaurant</span>
            </h2>

            <p className="reveal-item text-[var(--text-secondary)] leading-relaxed">
              Step into a warm Italian atmosphere where tradition meets modern luxury,
              and every pizza is crafted with passion and perfection.
            </p>

            <p className="reveal-item text-[var(--text-secondary)] leading-relaxed text-sm">
              At Napoli Nero, every pizza is a crafted expression — baked in fire,
              refined with precision, and served with intention. Hand-selected ingredients,
              slow fermentation dough, and bold Italian flavors define our philosophy
              of understated excellence.
            </p>

            {/* Stats row */}
            <div className="reveal-item grid grid-cols-3 gap-4 py-4 border-y border-white/[0.06]">
              {[
                { value: '150+', label: 'Menu Items' },
                { value: '98%',  label: 'Happy Guests' },
                { value: '12',   label: 'Awards Won' },
              ].map((stat) => (
                <div key={stat.label} className="text-center">
                  <div className="font-display font-bold text-ember-400 text-2xl">{stat.value}</div>
                  <div className="text-[var(--text-muted)] text-[0.65rem] tracking-wider mt-1">{stat.label}</div>
                </div>
              ))}
            </div>

            <div className="reveal-item flex flex-wrap gap-3">
              <motion.button
                whileHover={{ scale: 1.03, y: -2 }}
                whileTap={{ scale: 0.97 }}
                className="btn btn-outline"
              >
                View Menu
              </motion.button>
              <motion.button
                whileHover={{ scale: 1.03, y: -2, boxShadow: '0 0 28px rgba(255,107,0,0.4)' }}
                whileTap={{ scale: 0.97 }}
                className="btn btn-primary"
              >
                Book a Table
              </motion.button>
            </div>
          </div>
        </div>
      </div>
    </section>
  );
}
